function getUrlParameterValue(url, parameter) {
var urlParamsAll = url.substring(url.indexOf("?" + 1));
var urlParamsArr = urlParamsAll.split("&");
for (let i = 0; i < urlParamsArr.length; i++) {
temp = urlParamsArr[i].split("=");
if (temp[0] == parameter) {
return temp[1];
}
}
return "Not Found";
}
